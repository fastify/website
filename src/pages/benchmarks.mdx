import HeroBanner from '@site/src/components/HeroBanner'
import BenchmarkInfo from '@site/src/components/BenchmarkInfo'

import benchmarksData from '@site/static/generated/benchmarks.json'

export function BenchmarkSection() {
  return benchmarksData.frameworks.map((framework) => (
    <BenchmarkInfo
      key={framework.name}
      name={framework.name}
      req={framework.requests}
      progressValue={framework.requests / benchmarksData.reference}
      test={framework.test}
      repository={framework.repository}
    />
  ))
}
